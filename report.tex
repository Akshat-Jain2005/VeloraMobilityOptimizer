\section{Introduction}

Corporate employee transport is a large-scale logistics problem: organisations must route a heterogeneous fleet of vehicles---sedans, vans, two-wheelers---to pick up employees within individual time windows and deliver them to a common office, all while minimising cost and respecting capacity, sharing preferences, and vehicle-type constraints.

\subsection{Problem Statement}

The \textbf{Heterogeneous Vehicle Routing Problem with Soft Time Windows (HVRPSTW)} generalises the classical VRPTW \cite{solomon1987} by introducing:
\begin{itemize}[nosep, leftmargin=1.2em]
    \item \textbf{Heterogeneous fleet:} vehicles differ in capacity, cost-per-km, speed, and fuel type \cite{dondo2007,golden2008}.
    \item \textbf{Soft time windows:} late arrivals incur a graduated penalty rather than hard infeasibility \cite{taillard1997}.
    \item \textbf{Priority-based tolerance:} high-priority employees receive tighter grace periods.
    \item \textbf{Sharing constraints:} employees may specify maximum co-passengers.
    \item \textbf{Vehicle preference:} employees may request premium vs.\ normal vehicles.
\end{itemize}

The problem is NP-hard; even the homogeneous VRPTW is NP-hard in the strong sense \cite{toth2014}. Our approach combines a multi-start constructive heuristic (Solomon I1) with Simulated Annealing, achieving \textbf{74--87\% cost savings} across all test cases within a 24-second wall-clock budget.

\section{Mathematical Formulation}

\subsection{Problem Structure and Graph Model}

We model the system as a directed graph $G = (V, A)$, where $V = D \cup P \cup \{c\}$.

\begin{instancebox}[Node Sets]
\begin{itemize}[nosep, leftmargin=1.2em]
    \item $D = \{d_1, d_2, \dots, d_m\}$\,---\,vehicle depots
    \item $P = \{1, 2, \dots, n\}$\,---\,employee pickup locations
    \item $c$\,---\,common drop-off (office)
\end{itemize}
\end{instancebox}

Each employee request $i \in P$ consists of a pickup at node $i$ and a drop-off at $c$. The heterogeneous fleet $K$ has vehicles differing in depot $d(k)$, capacity $Q_k$, cost-per-km $C_k$, speed, and availability time $A_k$. Each employee $i$ has a soft time window $[E_i, L_i]$ and priority $P_i \in \{1,\dots,5\}$.

\subsection{Decision Variables}

\vspace{-4pt}
\begin{itemize}[nosep, leftmargin=1.2em]
    \item $x_{uv}^k \in \{0,1\}$: 1 if vehicle $k$ traverses arc $(u,v)$
    \item $t_u^k \ge 0$: arrival time of vehicle $k$ at node $u$
    \item $q_u^k \ge 0$: passengers onboard after leaving $u$
\end{itemize}

\subsection{Objective Function}

\begin{equation}
\min \; Z = w_c \cdot Z_{\text{dist}} + w_t \cdot Z_{\text{time}} + Z_{\text{late}} + Z_{\text{unassigned}}
\end{equation}

\begin{minipage}[t]{0.48\textwidth}
\textbf{Operating Cost}
\begin{equation}
Z_{\text{dist}} = \sum_{k \in K} \sum_{(u,v)\in A} C_k \, d_{uv} \, x_{uv}^k
\end{equation}

\textbf{Operating Time}
\begin{equation}
Z_{\text{time}} = \sum_{k \in K} T_k,\; T_k = \sum_{(u,v) \in A} \tau_{uv} x_{uv}^k
\end{equation}
\end{minipage}
\hfill
\begin{minipage}[t]{0.48\textwidth}
\textbf{Unassigned Penalty}
\begin{equation}
Z_{\text{unassigned}} = M \cdot |\{i \in P : i \text{ unassigned}\}|
\end{equation}

\textbf{Total Lateness Penalty}
\begin{equation}
Z_{\text{late}} = \sum_{i \in P} \text{Penalty}_i
\end{equation}
\end{minipage}

\subsubsection{Two-Tier Lateness Penalty}

\begin{equation}
\text{Penalty}_i = 
\begin{cases}
0 & \text{if } t_c^k \leq L_i + f(P_i) \\[3pt]
(t_c^k - L_i - f(P_i)) \cdot M \cdot w(P_i) & \text{otherwise}
\end{cases}
\end{equation}

\begin{insightbox}[Tolerance Configuration]
\small
\begin{tabular}{lccccc}
\toprule
\textbf{Priority} & P1 & P2 & P3 & P4 & P5 \\
\midrule
\textbf{Tolerance $f(P_i)$} & 5\,min & 10\,min & 15\,min & 20\,min & 30\,min \\
\textbf{Weight $w(P_i)$} & 5 & 4 & 3 & 2 & 1 \\
\bottomrule
\end{tabular}

\vspace{2pt}
\textbf{No penalty} while $t_c^k \le L_i + f(P_i)$. Exceeding tolerance triggers severe multiplier $M = 1000$.
\end{insightbox}

\subsection{Constraints}

\begin{minipage}[t]{0.48\textwidth}
\textbf{Pickup Assignment}---at most one vehicle:
\begin{equation}
\sum_{k\in K}\sum_{v\in V} x_{iv}^k \le 1 \;\;\forall i \in P
\end{equation}

\textbf{Flow Conservation:}
\begin{equation}
\sum_{v} x_{uv}^k - \sum_{v} x_{vu}^k = 0 \;\;\forall u \neq d(k),c
\end{equation}

\textbf{Route Structure:} each vehicle starts from depot $d(k)$, ends at $c$.
\end{minipage}
\hfill
\begin{minipage}[t]{0.48\textwidth}
\textbf{Capacity:}
\begin{align}
0 \le q_u^k &\le Q_k \\
q_v^k &\ge q_u^k + \delta(v) - (1{-}x_{uv}^k)M'
\end{align}

\textbf{Time Propagation:}
\begin{equation}
t_v^k \ge t_u^k + \tau_{uv} - (1{-}x_{uv}^k)M''
\end{equation}

\textbf{Vehicle Availability:}
\begin{equation}
t_{d(k)}^k \ge A_k \;\;\forall k\in K
\end{equation}
\end{minipage}

\vspace{4pt}
\textbf{Pickup Time Window:}\quad $t_i^k \ge E_i - (1 - \sum_{v} x_{iv}^k) \cdot M \;\; \forall i \in P,\; \forall k \in K$

\textbf{Sharing:} Each employee $i$ specifies max co-passengers $S_i$; violations penalised as $Z_{\text{sharing}} = \sum_i \max(0,\, |\text{co-riders}(i)| - S_i) \times M_{\text{share}}$. \textbf{Vehicle preference:} Mismatches penalised as $Z_{\text{pref}} = M_{\text{pref}} \cdot |\{i : \text{cat}(k(i)) \neq \text{pref}(i)\}|$.

\textbf{Distance:} Adjusted Haversine: $D_{\text{road}} \approx 1.4 \times D_{\text{haversine}}$ (accounts for urban road geometry).

\section{Algorithm Design}

The solver implements a two-phase hybrid: constructive heuristic $\to$ Simulated Annealing metaheuristic.

\subsection{Phase 1: Multi-Start Constructive Insertion (Solomon I1)}

Requests are sorted by priority and time urgency; across multiple restarts the order is perturbed to diversify constructions \cite{solomon1987,cordeau2002}. For vehicle $k$ with route $R_k$ and request $u$, we evaluate insertion at index $(i,j)$:

\begin{equation}
\Delta(u,k,i,j) = w_c\,\Delta\text{CTC}(u,k,i,j) + w_t\,\Delta\text{Time}(u,k,i,j) + \Delta\text{Penalty}(u,k,i,j)
\end{equation}

Restarts adapt to size: $R = \max(1,\min(10,\lfloor 50/\lceil 2N/V \rceil\rfloor))$. The best insertion across all feasible positions and vehicles minimises $\Delta$. Infeasible requests incur penalty $M = 100{,}000$.

\subsection{Phase 2: Simulated Annealing}

Given current solution $S$ and neighbour $S'$ with $\Delta E = Z(S') - Z(S)$ \cite{kirkpatrick1983}:

\begin{equation}
\Pr(\text{accept}) =
\begin{cases}
1 & \Delta E < 0 \\
\exp(-\Delta E / T) & \Delta E \ge 0
\end{cases}
\end{equation}

\begin{insightbox}[SA Hyperparameters (Tuned)]
\small
\begin{tabular}{llp{7cm}}
\toprule
\textbf{Parameter} & \textbf{Value} & \textbf{Rationale} \\
\midrule
\rowcolor{rowalt}
$T_0$ & $\min(0.3 \times Z_{\text{init}},\; 1000)$ & Adaptive to instance; cap prevents wild acceptance in penalty-heavy cases \\
$\alpha$ & 0.99 & Geometric cooling: $T_{r+1} = \alpha\, T_r$ \\
\rowcolor{rowalt}
$T_{\min}$ & 1.0 & Floor temperature \\
Wall-clock limit & 24\,s & Hard deadline for production deployment \\
\bottomrule
\end{tabular}
\end{insightbox}

\subsection{Time Complexity}

\[
T = \underbrace{O(N \log N)}_{\text{sort}}
  + \underbrace{O\!\left(\frac{R \cdot N^3}{V}\right)}_{\text{construction}}
  + \underbrace{O\!\left(I \cdot \frac{N^2}{V^2}\right)}_{\text{SA}}
  + \underbrace{O(N^2)}_{\text{distance matrix}}
\]

\begin{resultbox}
For $N \gg V$ (typical: 100 employees, $<$\,10 vehicles):\quad
$\displaystyle\boxed{O\!\left(\frac{N^3}{V}\right)}$\quad---\quad sub-second for $N \le 30$, $<$\,2s for $N = 100$.
\end{resultbox}

The React application (Vite) is deployed as a \textbf{Render Static Site}: build command \code{npm install \&\& npm run build}, publish directory \code{frontend/dist}. \subsection{TC04 --- 10 Employees, 4 Vehicles (Detailed Analysis)}

TC04 is the most complex test case, designed to stress every dimension of the HVRPSTW simultaneously: a fully heterogeneous fleet, mixed priorities (P1--P4), tight time windows, solo-ride sharing constraints, vehicle-preference requirements, and one deliberately infeasible outlier employee (E10). We analyse TC04 under two configurations---\textbf{penalties active} (soft constraints enforced) and \textbf{penalties off} (pure cost minimisation)---to isolate the solver's constraint-handling behaviour.

\subsubsection{Input Structure --- What Each Field Represents}

The input JSON (\filename{tc04\_input.json}) comprises four top-level objects:

\begin{enumerate}[nosep, leftmargin=1.2em]
    \item \textbf{\code{config}} --- Global solver parameters:
    \begin{itemize}[nosep, leftmargin=1em]
        \item \code{weights}: objective function split (\code{cost}: 0.7, \code{time}: 0.3).
        \item \code{tolerances}: per-priority grace periods $\{$P1: 5\,min, P2: 10, P3: 15, P4: 20, P5: 30$\}$.
        \item \code{allow\_external\_maps}: enables OSRM API for real road distances (fallback: Haversine $\times$ 1.4).
    \end{itemize}
    \item \textbf{\code{vehicles}} --- Fleet of 4 heterogeneous vehicles (Table~\ref{tab:tc04-fleet}).
    \item \textbf{\code{requests}} --- 10 employee pickup/drop-off requests (Table~\ref{tab:tc04-requests}).
    \item \textbf{\code{baseline}} --- Individual taxi costs per employee (total \rupee\,4,420 / 485\,min); used to compute savings.
\end{enumerate}

\section{Conclusion \& Future Work}

\subsection{Summary}

\textbf{Velora Mobility Optimizer} addresses the HVRPSTW with a two-phase hybrid solver (Solomon~I1 construction + Simulated Annealing), achieving \textbf{74--87\% cost savings} on instances of 8--15 employees within a 24-second wall-clock budget. Key contributions:
\begin{enumerate}[nosep, leftmargin=1.2em]
    \item Priority-aware two-tier penalty model with configurable tolerance per priority level.
    \item Five weighted neighbourhood operators (intra- and inter-route) for diversified search.
    \item Adaptive hyperparameters ($T_0$ proportional to initial cost, capped at 1000).
    \item Docker-containerised deployment on Render with React web dashboard and Flutter mobile app.
    \item All five adversarial hack tests passed---infeasibility detection, scalability, and spatial intelligence validated.
\end{enumerate}

\subsection{Limitations}

\begin{itemize}[nosep, leftmargin=1.2em]
    \item Structurally infeasible outliers (e.g.\ E10) cannot be resolved by any solver---correctly reported with penalty.
    \item Zero-penalty mode causes aggressive consolidation, leading to widespread sharing and preference violations.
    \item Single-threaded SA; parallel neighbourhood evaluation would improve runtime for $N > 50$ instances.
    \item Static one-shot routing; no real-time re-routing for live traffic or last-minute cancellations.
\end{itemize}

\subsection{Future Work}

\begin{recbox}[Planned Enhancements]
\begin{enumerate}[nosep, leftmargin=1.2em]
    \item \textbf{Parallel SA:} OpenMP neighbourhood evaluation (2--4$\times$ speedup).
    \item \textbf{Adaptive operator weights:} ALNS-style dynamic probability adjustment.
    \item \textbf{Real-time re-routing:} WebSocket live updates; incremental SA from current best.
    \item \textbf{OSRM integration:} Replace Haversine $\times 1.4$ with actual road distances.
    \item \textbf{Flutter app completion:} Implement designed 4-screen app; publish to app stores.
\end{enumerate}
\end{recbox}
